/* ===== CONTROL DEL TECLADO ===== */ document.addEventListener("keydown", function(event){ let tecla = event.key; /* NUMEROS */ if(!isNaN(tecla)){ agregar(tecla); } /* OPERADORES */ if(tecla === "+"){ agregar("+"); } if(tecla === "-"){ agregar("-"); } if(tecla === "*"){ agregar("*"); } if(tecla === "/"){ agregar("/"); } /* DECIMAL */ if(tecla === "."){ agregar("."); } /* ENTER = CALCULAR */ if(tecla === "Enter"){ calcular(); } /* BACKSPACE = BORRAR */ if(tecla === "Backspace"){ borrar(); } /* ESCAPE O C = LIMPIAR */ if(tecla === "Escape" || tecla === "c" || tecla === "C"){ limpiar(); } });
